React native - Loading Page

Sometimes the page is loading, we will keep a loading page to give user more user experience, how it works? We can use “Modal” and “ActivityIndicator” do it very well.

Let us start:

import {
    ActivityIndicator,
    Modal
} from 'react-native';

export default class Test extends React.Component {

    render() {
        return (
            <Fragment>
                <StatusBar barStyle="dark-content" />
                <SafeAreaView style={styles.container}>
                     <Modal
                        transparent={true}
                        animationType={'fade'}
                        visible={this.state.isLoading}
                        onRequestClose={() => {console.log('close modal')}}>
                        <View style={styles.modalBackground}>
                            <View style={styles.activityIndicatorWrapper}>
                            <ActivityIndicator style={styles.ActivityIndicatorStyle} />
                            </View>
                        </View>
                    </Modal>
                </SafeAreaView>
            </Fragment>
        )
    }
}

const styles = StyleSheet.create({
    container: {
        flex: 1,
        flexDirection: 'column',
        height: '100%',
        width: '100%',
        backgroundColor: '#F3F3F3',
        alignItems: 'center',
    },
      activityIndicatorWrapper: {
        backgroundColor: '#FFFFFF',
        height: 70,
        width: 70,
        borderRadius: 10,
        display: 'flex',
        alignItems: 'center',
        justifyContent: 'space-around'
      },
      modalBackground: {
        flex: 1,
        alignItems: 'center',
        flexDirection: 'column',
        justifyContent: 'space-around',
        backgroundColor: '#00000040'
      },
      ActivityIndicatorStyle: {
      	position: 'absolute', 
      	left: 0, 
      	right: 0, 
      	top: 0, 
      	bottom: 0, 
      	margin: "auto"
      }
});
